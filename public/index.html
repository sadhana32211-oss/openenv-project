API Documentation

POST /api/reset Reset the environment to initial state
POST /api/step Take an action: { action: 0-3 }
GET /api/state Get current state without taking action
GET /api/env/info Get environment information

Usage Example


// Using the OpenEnv API

// 1. Reset the environment
const resetResponse = await fetch('/api/reset', { method: 'POST' });
const { state } = await resetResponse.json();
console.log('Initial state:', state);

// 2. Take actions
const actions = { UP: 0, RIGHT: 1, DOWN: 2, LEFT: 3 };

const stepResponse = await fetch('/api/step', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ action: actions.RIGHT })
});

const { state: newState, reward, done, info } = await stepResponse.json();
console.log('Reward:', reward, 'Done:', done);

// 3. Get current state
const stateResponse = await fetch('/api/state');
const currentState = await stateResponse.json();
console.log('Current state:', currentState);

// 4. Training loop example
async function train() {
  await fetch('/api/reset', { method: 'POST' });
  
  let done = false;
  let totalReward = 0;
  
  while (!done) {
    const action = agent.selectAction(); // Your RL agent
    const response = await fetch('/api/step', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ action })
    });
    
    const { reward, done: episodeDone, state } = await response.json();
    totalReward += reward;
    done = episodeDone;
    
    agent.learn(state, reward); // Update your agent
  }
  
  console.log('Episode complete! Total reward:', totalReward);
}
